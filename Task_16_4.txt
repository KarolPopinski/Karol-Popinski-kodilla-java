USE kodilla_course;

SHOW TABLES;

SELECT ISSUESLISTS.NAME AS TASK_STATUS, ISSUES.DESCRIPTION AS TASK_DESCRIPTION
FROM ISSUESLISTS
JOIN ISSUES ON ISSUESLISTS.ID = ISSUES.ISSUESLIST_ID;

SELECT ISSUES.DESCRIPTION AS TASK_NAME, USERS.ID AS USER_NUMBER
FROM ISSUES
JOIN USERS ON ISSUES.USER_ID_ASSIGNEDTO = USERS.ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS NUMBER_OF_TASKS
FROM USERS U JOIN ISSUES ON U.ID = ISSUES.USER_ID_ASSIGNEDTO
GROUP BY USER_ID_ASSIGNEDTO;
